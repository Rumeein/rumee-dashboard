// FULL LOCKDOWN + ROLES (dashboard memory active.md #40/#41, 2026-07-11) --
// every collection in this project now requires a signed-in, PROVISIONED
// user (a doc in rumee_users) to read OR write. Supersedes the 2026-07-02/
// 07-05 "scoped to product_master/Purchases only, everything else stays
// public" design below -- Jaiswal's explicit choice: the whole dashboard
// should require sign-in, not just individual tabs, and more than one
// person (owner + staff) needs to be able to sign in.
//
// DO NOT PUBLISH THIS FILE YET. All of these must be true first:
//   1. rumee_users/rumeein@gmail.com (role: 'owner') has been seeded via
//      `python process.py --seed-users` (Admin SDK, bypasses these rules to
//      create it) -- otherwise NOBODY, including Jaiswal, would pass
//      isOwner() the moment this publishes, since isOwner() now depends on
//      that doc existing instead of a hardcoded email.
//   2. index.html's DASHBOARD_LOGIN_REQUIRED flag (BOOT section) is flipped
//      to true and pushed live -- until then the dashboard UI never signs
//      anyone in, so publishing these rules first would just break the live
//      public site (every read starts failing with no login screen to fix it).
//   3. Vantage's context_builder.py has been migrated from plain public
//      API-key reads to the Firebase Admin SDK (it already holds a working
//      service-account credential -- used today only for writing
//      rumee_insights/rumee_tasks -- that needs extending to reads too).
//      Admin SDK access always bypasses these rules regardless of how
//      strict they are, so once done Vantage is unaffected by anything
//      below. Until done, publishing this file breaks Vantage's daily
//      brief. See active.md #40 for the exact Vantage-session prompt used
//      to kick this off -- that work happens in a separate Vantage session,
//      not this repo.
// Pipeline writes (process.py, GitHub Actions) always use the Admin SDK
// (service account) and always bypass these rules regardless -- none of
// this affects the pipeline in any way, only client requests using the
// public FB_API_KEY (the dashboard itself, and Vantage until migrated).
// ═══════════════════════════════════════════════════════════════════════

service cloud.firestore {
  match /databases/{database}/documents {

    // Role lookup replaces the old hardcoded-email isOwner() (2026-07-05 -
    // 07-11) with a data-driven one, so more than one person can be an
    // owner and a second role (staff) can exist. Doc id = the signed-in
    // user's own email (not their uid) -- lets Jaiswal provision someone
    // by typing their email BEFORE that person has ever signed in once.
    function userRole() {
      return get(/databases/$(database)/documents/rumee_users/$(request.auth.token.email)).data.role;
    }
    function isOwner() {
      return request.auth != null && userRole() == 'owner';
    }
    // Staff can enter Purchases/Materials data but nothing else (Jaiswal,
    // 2026-07-11) -- applied only to the specific collections below, never
    // as a blanket replacement for isOwner().
    function isStaffOrOwner() {
      return request.auth != null && (userRole() == 'owner' || userRole() == 'staff');
    }

    match /rumee_users/{userId} {
      // Anyone signed in can read their OWN role doc -- this is what makes
      // userRole()'s get() above resolve without circularity: a user
      // checking their own doc always matches the first clause directly,
      // never needs isOwner() to already be true. Reading anyone ELSE's
      // entry (the Manage Users list) requires actually being an owner.
      allow read: if request.auth != null && (request.auth.token.email == userId || isOwner());
      // Only an owner can grant/change access, and never by removing their
      // OWN owner doc (guards against a self-lockout with no other owner
      // provisioned yet) -- the UI also blocks this, this is the real
      // enforcement boundary.
      allow create, update: if isOwner();
      allow delete: if isOwner() && userId != request.auth.token.email;
    }

    // Decided 2026-07-02 (see DOCS.md §22 Decisions): protects the top-level
    // scalar fields the dashboard is allowed to touch. Firestore cannot
    // validate individual fields inside an array element, so 'listings' is
    // restricted to append-only (size can't shrink) rather than field-level
    // — accepted gap, since the dashboard's only write into listings[] is an
    // arrayUnion append via the Assign flow, never an edit of existing entries.
    // isOwner() added 2026-07-11 (full lockdown) on top of the original
    // field-level restrictions below -- those restrictions still apply on
    // top of, not instead of, requiring a signed-in owner.
    match /product_master/{docId} {
      allow read: if isOwner();

      allow create: if false; // pipeline (Admin SDK) creates docs, not the public client

      allow update: if isOwner() &&
        // sku_id and created_at can never be touched by a public write
        !request.resource.data.diff(resource.data).affectedKeys().hasAny(['sku_id', 'created_at'])
        // only these top-level fields may change
        && request.resource.data.diff(resource.data).affectedKeys()
             .hasOnly(['design', 'variation_type', 'status', 'fk_url', 'notes', 'listings'])
        // listings, if touched, may only grow (append-only — no edit/remove of existing entries)
        && (
          !request.resource.data.diff(resource.data).affectedKeys().hasAny(['listings'])
          || request.resource.data.listings.size() >= resource.data.listings.size()
        );

      allow delete: if false;
    }

    match /needs_review/{docId} {
      allow read: if isOwner();
      allow create: if false; // pipeline (Admin SDK) only

      allow update: if false;

      // Delete only allowed once the matching pm_overrides doc exists —
      // i.e. only as the LAST step of a real Assign flow, not a free-standing
      // public delete. needs_review id is 'nr_{platform}_{catalog_id}';
      // pm_overrides id is 'nr_' stripped, i.e. '{platform}_{catalog_id}'.
      // Added 2026-07-02 after security review — closes an unpaired public
      // delete gap on this collection.
      allow delete: if isOwner() && exists(/databases/$(database)/documents/pm_overrides/$(docId[3:]));
    }

    match /pm_overrides/{docId} {
      allow read: if isOwner();

      // Create only allowed if a matching needs_review doc exists for this
      // exact platform_catalog_id — prevents an anonymous client from
      // fabricating an arbitrary catalog_id -> target_sku_id mapping that
      // the pipeline would otherwise trust unconditionally on its next run.
      // Added 2026-07-02 after security review.
      allow create: if isOwner() && exists(/databases/$(database)/documents/needs_review/$('nr_' + docId));

      allow update: if false;
      allow delete: if false;
    }

    // Purchases module (DOCS.md §26 / dashboard memory active.md #25).
    // First browser-authenticated write path in this codebase — added
    // 2026-07-05 alongside enabling Firebase Authentication (Google
    // Sign-In). READ was already restricted here from day one (not just
    // write), before the rest of the project caught up on 2026-07-11.
    match /rumee_purchases/{docId} {
      // Staff can enter/view Purchases (Jaiswal, 2026-07-11) -- this is the
      // one collection group staff access was explicitly asked for.
      allow read: if isStaffOrOwner();

      // New purchases are always entered active — void is a later action,
      // never the initial state. entered_by must match the caller's own
      // token (added after security review, 2026-07-11) -- without this a
      // staff account could forge entered_by to impersonate the owner in
      // the audit trail, since the field was previously only set client-side
      // with nothing stopping a direct API write from lying about it.
      allow create: if isStaffOrOwner()
        && request.resource.data.status == 'active'
        && request.resource.data.entered_by == request.auth.token.email;

      // Two independent update paths (Jaiswal asked for full edit access on
      // 2026-07-07/08, on top of the original void+re-enter design):
      //
      // (A) Narrow path — payment_status/amount_paid (a bill can legitimately
      // receive more payment later, or Amount Paid needs correcting),
      // status (toggles between active and void only — void must be
      // reversible, financial records are never deleted, 2026-07-07), and
      // qc_events (QC happens gradually, new events append over time).
      //
      // (B) Full-edit path — any other header/detail field (vendor, date,
      // category, GST/packing/courier/other_transport, invoice/GSTIN,
      // payment_mode, notes, details[]) may change via the "Edit" button,
      // but this path is NOT allowed to touch status or qc_events in the
      // same write — those stay governed only by path (A)'s dedicated
      // actions (Void/Restore, QC entry), so a full edit can never
      // accidentally flip status or tamper with QC history. purchase_id and
      // entered_by must also stay unchanged here -- the 2026-07-12
      // independent review caught that this path previously had no field
      // restriction beyond status/qc_events, so a staff/owner edit could
      // silently rewrite entered_by (or purchase_id) and defeat the same
      // audit-trail protection create() already enforces.
      allow update: if isStaffOrOwner()
        && request.resource.data.entered_by == resource.data.entered_by
        && request.resource.data.purchase_id == resource.data.purchase_id
        && (
          (
            request.resource.data.diff(resource.data).affectedKeys()
              .hasOnly(['payment_status', 'amount_paid', 'status', 'qc_events'])
            && (
              !request.resource.data.diff(resource.data).affectedKeys().hasAny(['status'])
              || (resource.data.status == 'active' && request.resource.data.status == 'void')
              || (resource.data.status == 'void' && request.resource.data.status == 'active')
            )
            && (
              !request.resource.data.diff(resource.data).affectedKeys().hasAny(['qc_events'])
              || request.resource.data.qc_events.size() >= resource.data.qc_events.size()
            )
          )
          ||
          (
            !request.resource.data.diff(resource.data).affectedKeys().hasAny(['status', 'qc_events'])
          )
        );

      allow delete: if false; // corrections go through status=void, never a real delete
    }

    match /rumee_payees/{docId} {
      // Vendor + job-worker picker list (active.md #25) — small reference
      // data, no embedded arrays, no in-place-edit concerns. Staff needs
      // this to enter Purchases (vendor/job-worker picker), same as owner.
      // delete: if false added after security review, 2026-07-11 -- plain
      // "write" implicitly grants delete too, which contradicts this
      // collection's own void/restore convention (voidPayee/restorePayee
      // only ever call update, never delete) and would let a staff account
      // permanently erase a payee via a direct API call the UI never offers.
      allow read, create, update: if isStaffOrOwner();
      allow delete: if false;
    }

    match /rumee_pur_categories/{docId} {
      // Purchase line-item category list (Jaiswal, 2026-07-08) — Earring/Raw
      // Material/Packaging/Other plus whatever he adds, each with its own
      // qc_enabled/design_id_enabled flags. Same small-reference-collection
      // shape and staff-or-owner access as rumee_payees, same delete lockdown.
      allow read, create, update: if isStaffOrOwner();
      allow delete: if false;
    }

    // BOM / Materials costing system (dashboard memory active.md #39,
    // 2026-07-11) — raw material / component master, job-work conversion
    // batches, and the stock movement ledger. Staff can enter this too
    // (Jaiswal, 2026-07-11 — e.g. a helper logging a job-work batch).
    // delete: if false + entered_by binding added after security review,
    // 2026-07-11 -- same two gaps as rumee_purchases above: plain "write"
    // implicitly allowed delete (contradicting voidMaterial/voidConversion's
    // own void-not-delete design, and erasing the cost/stock audit trail
    // with no client UI ever exposing that action), and entered_by was
    // trusted from the client with nothing stopping a forged value.
    match /rumee_materials/{docId} {
      allow read: if isStaffOrOwner();
      // entered_by locked to whoever created the doc -- update alone had no
      // field restriction at all until the 2026-07-12 independent review
      // caught it (a staff/owner update could silently rewrite entered_by,
      // defeating the same audit-trail protection create() enforces).
      allow update: if isStaffOrOwner() && request.resource.data.entered_by == resource.data.entered_by;
      allow create: if isStaffOrOwner() && request.resource.data.entered_by == request.auth.token.email;
      allow delete: if false;
    }
    match /rumee_conversions/{docId} {
      allow read: if isStaffOrOwner();
      allow update: if isStaffOrOwner() && request.resource.data.entered_by == resource.data.entered_by;
      allow create: if isStaffOrOwner() && request.resource.data.entered_by == request.auth.token.email;
      allow delete: if false;
    }
    match /rumee_stock_ledger/{docId} {
      // Append-only audit trail -- no update either, only create (+read).
      allow read: if isStaffOrOwner();
      allow create: if isStaffOrOwner() && request.resource.data.entered_by == request.auth.token.email;
      allow update: if false;
      allow delete: if false;
    }

    // Recipes (BOM ratios, dashboard memory active.md #44/#45, 2026-07-12) --
    // one doc per output material, id = that material's own material_id.
    // Created the first time a Conversion Batch is run for a new output
    // (ratio derived from that real batch's actual quantities, never a
    // separate blank-form entry) and only updated again when Jaiswal
    // explicitly opts in via a later batch. Same staff-or-owner access and
    // entered_by binding as the collections above -- not a financial record
    // itself, but directly drives derived cost, same sensitivity class.
    match /rumee_recipes/{docId} {
      allow read: if isStaffOrOwner();
      allow update: if isStaffOrOwner() && request.resource.data.entered_by == resource.data.entered_by;
      allow create: if isStaffOrOwner() && request.resource.data.entered_by == request.auth.token.email;
      allow delete: if false;
    }

    // Every other collection the dashboard/pipeline/Vantage use — as of
    // 2026-07-11 these are ALSO owner-only (full lockdown, active.md #40).
    // Explicitly enumerated instead of a blanket wildcard for the same
    // reason as before: Firestore ORs every matching `match` block, so a
    // trailing `match /{document=**} { allow ...: if true }` would silently
    // override every specific restriction above regardless of order
    // (confirmed via Rules Playground, 2026-07-05 incident) — deny-by-
    // default (explicit list + `if false` catch-all) is the only way
    // specific rules above can coexist with a shared fallback.
    //
    // Vantage (brief_builder.py/context_builder.py) reads every one of
    // these via the same plain public-API-key mechanism the dashboard used
    // to use — until it's migrated to Admin SDK reads (see the file-level
    // comment at the top), publishing this version of the file breaks
    // Vantage's daily brief. Do not publish until that migration is done.
    match /rumee_db/{docId}               { allow read, write: if isOwner(); }
    match /rumee_insights/{docId}         { allow read, write: if isOwner(); }
    match /rumee_tasks/{docId}            { allow read, write: if isOwner(); }
    match /rumee_settings/{docId}         { allow read, write: if isOwner(); }
    match /rumee_fk_daily/{docId}         { allow read, write: if isOwner(); }
    match /rumee_me_daily/{docId}         { allow read, write: if isOwner(); }
    match /rumee_fk_returns_daily/{docId} { allow read, write: if isOwner(); }
    match /rumee_fk_ads_daily/{docId}     { allow read, write: if isOwner(); }
    match /rumee_fk_ads_kw/{docId}        { allow read, write: if isOwner(); }
    match /rumee_fk_ads_sku/{docId}       { allow read, write: if isOwner(); }
    match /rumee_fk_ads_placements/{docId}  { allow read, write: if isOwner(); }
    match /rumee_fk_ads_order_items/{docId} { allow read, write: if isOwner(); }
    match /rumee_me_ads_daily/{docId}     { allow read, write: if isOwner(); }
    match /rumee_me_ads_catalog/{docId}   { allow read, write: if isOwner(); }
    match /rumee_keywords/{docId}         { allow read, write: if isOwner(); }
    match /rumee_az_catalog/{docId}       { allow read, write: if isOwner(); }
    match /rumee_az_daily/{docId}         { allow read, write: if isOwner(); }
    match /rumee_orders_daily/{docId}     { allow read, write: if isOwner(); }
    match /rumee_orders_sku/{docId}       { allow read, write: if isOwner(); }

    // Deny by default. Anything not explicitly matched above (including
    // any future new collection someone forgets to add here) is closed,
    // not silently open. Add a new match block above whenever a new
    // collection is introduced.
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
